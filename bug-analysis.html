BUG #3: broadcastToClients() sem try-catch

Local: multiplayer.js - broadcastToClients()

// CÓDIGO SEM PROTEÇÃO:
function broadcastToClients(data) {
    connectedClients.forEach(client => {
        if (client.open) client.send(data);  // ❌ Pode lançar exceção!
    });
}
// CORREÇÃO:
function broadcastToClients(data) {
    connectedClients.forEach(client => {
        if (!client || !client.open) return;
        try {
            client.send(data);
        } catch(e) {
            console.error('Erro ao enviar:', e);
        }
    });
}
