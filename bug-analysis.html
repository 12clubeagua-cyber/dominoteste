#1: input.js - highlight() - Esconder picker no início

// highlight() - ADICIONAR NO INÍCIO:
function highlight(moves) {
  // ✅ ESCONDER PICKER SEMPRE QUE HIGHLIGHT FOR CHAMADO
  const picker = document.getElementById('side-picker');
  if (picker) picker.style.display = 'none';
  
  // Limpar highlights anteriores
  document.querySelectorAll('.tile').forEach(el => el.classList.remove('playable'));
  removePlayableListeners();

  moves.forEach(x => {
    const el = document.getElementById(`my-tile-${x.idx}`);
    if (!el) return;
    el.classList.add('playable');
    el.onclick = () => {
      if (typeof safeAudioInit === 'function') safeAudioInit();
      if (STATE.isBlocked) return;

      // ✅ VERIFICAR SE PRECISA PICKER ANTES DE QUALQUER COISA
      const extremesAreDifferent = STATE?.extremes?.[0] !== STATE?.extremes?.[1];
      const needsPicker = x.side === 'both' && extremesAreDifferent && STATE.positions?.length > 0;
      
      console.log('[PICKER] needsPicker:', needsPicker, 'side:', x.side, 'extremes:', STATE?.extremes);

      STATE.isBlocked = true;

      // Remove visual
      document.querySelectorAll('.tile.playable').forEach(tile => tile.classList.remove('playable'));
      const hand0 = document.getElementById('hand-0');
      if (hand0) hand0.classList.remove('active-turn');
      removePlayableListeners();
      void document.body.offsetHeight; 

      requestAnimationFrame(() => {
          if (needsPicker) {
            STATE.pendingIdx = x.idx;
            if (picker) {
                picker.style.display = 'flex';
                console.log('[PICKER] Picker mostrado!');
            }
          } else {
            const side = (x.side === 'both' || x.side === 'any') ? 0 : x.side;
            play(myPlayerIdx, x.idx, side);
          }
      });
    };
  });
}

📊 QUANDO O PICKER DEVE APARECER?

O picker só aparece se TODAS estas condições forem verdadeiras:

const needsPicker = 
    x.side === 'both'           // ✅ Peça encaixa nos DOIS lados
    && extremesAreDifferent     // ✅ Extremidades SÃO DIFERENTES (ex: 3 e 5)
    && STATE.positions?.length > 0;  // ✅ JÁ TEM PEÇAS NA MESA

// EXEMPLOS:
// 1. Primeira peça do jogo (STATE.positions.length = 0)
//    → needsPicker = false  ✅ (peça vai em qualquer lugar)
//
// 2. Extremidades iguais (ex: ambos 4)
//    → needsPicker = false  ✅ (não importa qual lado)
//
// 3. Peça só encaixa em um lado (side = 0 ou side = 1)
//    → needsPicker = false  ✅ (já sabe onde vai)
//
// 4. Peça encaixa nos dois lados E extremidades diferentes E tem peças na mesa
//    → needsPicker = true   ✅ (MOSTRAR PICKER!)

📈 FLUXO CORRETO DO PICKER

INÍCIO DA PARTIDA:
  ↓
processTurn() → highlight()
  ↓
highlight(): picker.style.display = 'none'  ✅ (escondido)
  ↓
Jogador clica na peça
  ↓
onclick: Verifica needsPicker
  ↓
SE needsPicker = false:
  → play() → animação → próximo turno
  → picker continua escondido  ✅
  ↓
SE needsPicker = true:
  → picker.style.display = 'flex'  ✅ (mostrado!)
  → Jogador clica em Cima/Baixo/Cancelar
  → executeMove() ou cancelMove()
  → picker.style.display = 'none'  ✅ (escondido)
  → play() ou highlight()
  ↓
PRÓXIMO TURNO (não é meu):
  ↓
processTurn() → renderHands()
  ↓
renderHands(): picker.style.display = 'none'  ✅ (escondido)
  ↓
PRÓXIMO TURNO (é meu):
  ↓
processTurn() → highlight()
  ↓
highlight(): picker.style.display = 'none'  ✅ (escondido até clicar)
